Hint for usage in the jrunscript console
The console object ist not defined by default. Before executing my JavaScript code you have to define this object and its log method. E.g.: 

        var console = {};
        console.log = function(msg) {
            var m = '';
            if (msg) {
                m = msg;
            }
            java.lang.System.out.println('' + m);
        }
